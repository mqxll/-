#include <iostream>
#include <cstdlib>
#include <ctime>
int main() {
    float** a = new float*[100];
    for(int i=0;i<100;i++) a[i]=new float[200];
    srand(time(0));
    for(int i=0;i<100;i++)
        for(int j=0;j<200;j++)
            a[i][j] = (rand()%20000-10000)/10000.0f;
    for(int i=0;i<100;i++) delete[] a[i];
    delete[] a;
}
