#include <fstream>
int main() {
    std::ofstream f("data.bin",std::ios::binary);
    for(int i=0;i<10000;i++){ int x=rand()%2?1:-1; f.write((char*)&x,sizeof(int)); }
    f.close();
    std::ifstream in("data.bin",std::ios::binary);
    int* arr=new int[10000];
    in.read((char*)arr,10000*sizeof(int));
    long long sum=0; for(int i=0;i<10000;i++) sum+=arr[i];
    delete[] arr;
}
